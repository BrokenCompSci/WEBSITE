def print_repeat(text):
    global print_text
    global print_count
    if text == print_text:
        print(("%1s" % "."))
        print_count = 1
    else:
        print_text = text
        if print_count > 0:
            print("")
            print_count = 0
            print(text)

    def clearTable(self):
        """ Clear table to empty
        """
        self.clearBalls()
        
    def draw_pockets(self):    # Draw pockets on table

        pocket_radius = self.scaleToPix(4)  # 2.54 cm
        offset = self.scaleToPix(0.5)  # 1.5 cm
        pockets = [ (offset, offset), (self.length / 2, 0), (self.length, offset),
            (offset, self.width), (self.length / 2, self.width), (self.length, self.width)
        ]
        for x, y in pockets:
            x_pix = self.scaleToPix(x)
            y_pix = self.scaleToPix(y)
            self.cvs.create_oval(
                x_pix - pocket_radius, y_pix - pocket_radius,
                x_pix + pocket_radius, y_pix + pocket_radius,
                fill="black"
            )
